### Task 3: Test pass triggers review schedule + P-value update

**Files:**
- Modify: `src/app/api/modules/[moduleId]/test/submit/route.ts:235-237` (isPassed block)

- [ ] **Step 1: Add review_schedule creation inside the isPassed block**

In `src/app/api/modules/[moduleId]/test/submit/route.ts`, find the `if (isPassed)` block inside the transaction (line ~235-237):

```typescript
    if (isPassed) {
      db.prepare('UPDATE modules SET learning_status = ? WHERE id = ?').run('completed', id)
    }
```

Expand it to:

```typescript
    if (isPassed) {
      db.prepare('UPDATE modules SET learning_status = ? WHERE id = ?').run('completed', id)

      // Create first review_schedule entry (skip if re-test and already exists)
      const existingSchedule = db.prepare(
        'SELECT id FROM review_schedule WHERE module_id = ? AND review_round = 1'
      ).get(id) as { id: number } | undefined

      if (!existingSchedule) {
        db.prepare(
          "INSERT INTO review_schedule (module_id, review_round, due_date, status) VALUES (?, 1, date('now', '+3 days'), 'pending')"
        ).run(id)
      }

      // Update cluster P-values based on test results
      // Join: test_questions.kp_id → knowledge_points.cluster_id → clusters
      const clusterResults = db.prepare(`
        SELECT
          kp.cluster_id,
          COUNT(*) as total,
          SUM(CASE WHEN tr.is_correct = 1 THEN 1 ELSE 0 END) as correct
        FROM test_questions tq
        JOIN knowledge_points kp ON tq.kp_id = kp.id
        JOIN test_responses tr ON tr.question_id = tq.id
        WHERE tq.paper_id = ?
        GROUP BY kp.cluster_id
      `).all(body.paper_id) as Array<{ cluster_id: number; total: number; correct: number }>

      for (const cr of clusterResults) {
        if (cr.cluster_id == null) continue

        const cluster = db.prepare(
          'SELECT current_p_value, consecutive_correct FROM clusters WHERE id = ?'
        ).get(cr.cluster_id) as { current_p_value: number; consecutive_correct: number } | undefined

        if (!cluster) continue

        if (cr.correct === cr.total) {
          // All correct: increment consecutive_correct, maybe upgrade P
          const newConsecutive = cluster.consecutive_correct + 1
          const newP = newConsecutive >= 2
            ? Math.min(cluster.current_p_value + 1, 5)
            : cluster.current_p_value
          db.prepare(
            'UPDATE clusters SET consecutive_correct = ?, current_p_value = ? WHERE id = ?'
          ).run(newConsecutive, newP, cr.cluster_id)
        } else {
          // Has errors: reset consecutive_correct, downgrade P
          const newP = Math.max(cluster.current_p_value - 1, 1)
          db.prepare(
            'UPDATE clusters SET consecutive_correct = 0, current_p_value = ? WHERE id = ?'
          ).run(newP, cr.cluster_id)
        }
      }
    }
```

**Key details:**
- The `clusterResults` query groups test questions by their knowledge point's cluster. `test_questions.kp_id` is always populated (set to `kpIds[0]` during generation — see `test/generate/route.ts:459`), so the JOIN covers all questions.
- The `kp_ids` JSON array column is NOT used here — `kp_id` (single) is sufficient for cluster mapping since it's always set.
- The duplicate guard (`existingSchedule`) prevents creating multiple round-1 entries on re-test.
- P-value bounds: min 1, max 5. Consecutive correct resets to 0 on any error.
